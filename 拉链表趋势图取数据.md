```sql
SELECT A
	.tt,
	dim_201_org_5mmf.full_name 
FROM
	(
	SELECT
		to_char(
			to_date(
				concat (
					'2021',
					'-',
					UNNEST ( ARRAY [ '01', '02', '03', '04', '05', '06', '07', '08', '09', '10', '11', '12' ] ) 
				),
				'yyyy-mm' 
			) + INTERVAL '1 month - 1s',
			'yyyymmddhh24miss' 
		) AS tt 
	) --A LEFT JOIN (select * from dwd.dim_201_org_5mmf where id = '2010000') dim_201_org_5mmf ON dim_201_org_5mmf.s_start_time <= A.tt
	A LEFT JOIN ( SELECT * FROM dwd.dim_201_org_5mmf WHERE ID = '2072100-0802' ) dim_201_org_5mmf ON dim_201_org_5mmf.s_start_time <= A.tt 
																																																		AND dim_201_org_5mmf.s_end_time >= A.tt 
ORDER BY
	A.tt
```
