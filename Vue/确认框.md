```vue
<a @click="handleStop(record)">停用</a>

methods: {
        // 停用供应商的确认框
        handleStop(record) {
            const _self = this;
            this.$confirm({
                title: '确认提示',
                content: `确认要停用该供应商吗？`,
                okType: 'danger',
                onOk() {
                    startOrStopSupplier(record.id, '0').then(res => {
                        // console.log('startOrStopSupplier-Rsp', res);
                        _self.confirmLoading = false;
                        _self.$refs.table.refresh();
                        const resultCode = res.resultCode;
                        if (resultCode.slice(resultCode.length - 4) === '0000') {
                            _self.$message.success(
                                '已停用供应商 【' + record.name + '】'
                            );
                        }
                    });
                }
            });
        },
}
```
